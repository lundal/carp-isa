                \optionalbitbox{8}{#2}
                \optionalbitbox{8}{#3}
                \bitbox{3}{#4} % length

\newcommand{\optionalbitbox}[2]{
    \ifthenelse{\equal{#2}{-}}
        {\unusedbitbox{#1}}
        {\bitbox{#1}{#2}}
}

\newcommand{\unusedbitbox}[1]{\bitbox{#1}{\color{lightgray}\rule{\width}{\height}}}
